  String _input = "";
  String _result = "0";

  final List<String> _buttons = [
    'C', '‚å´', '%', '√∑',
    '7', '8', '9', '√ó',
    '4', '5', '6', '‚àí',
    '1', '2', '3', '+',
    '0', '.', '=', 
  ];

  bool _isOperator(String x) {
    return ['√∑', '√ó', '‚àí', '+', '%'].contains(x);
  }

  void _onButtonClick(String text) {
    setState(() {
      if (text == 'C') {
        _input = "";
        _result = "0";
      } else if (text == '‚å´') {
        if (_input.isNotEmpty) {
          _input = _input.substring(0, _input.length - 1);
        }
      } else if (text == '=') {
        try {
          _result = _evaluate(_input);
        } catch (e) {
          _result = "Error";
        }
      } else {
        _input += text;
      }
    });
  }

  String _evaluate(String expression) {
    expression = expression
        .replaceAll('√ó', '*')
        .replaceAll('√∑', '/')
        .replaceAll('‚àí', '-');

    try {
      return _calculate(expression).toString();
    } catch (_) {
      return "Error";
    }
  }

  double _calculate(String exp) {
    try {
      final expClean = exp.replaceAll(' ', '');
      final parser = RegExp(r'([-]?\d+\.?\d*)([+\-*/%])?');
      final matches = parser.allMatches(expClean).toList();
      if (matches.isEmpty) return 0;

      double total = double.parse(matches[0].group(1)!);
      for (int i = 1; i < matches.length; i++) {
        final op = matches[i - 1].group(2);
        final next = double.parse(matches[i].group(1)!);
        switch (op) {
          case '+':
            total += next;
            break;
          case '-':
            total -= next;
            break;
          case '*':
            total *= next;
            break;
          case '/':
            total /= next;
            break;
          case '%':
            total %= next;
            break;
        }
      }
      return double.parse(total.toStringAsFixed(2));
    } catch (_) {
      return double.nan;
    }
  }

  Future<File?> _exportCardAsImage() async {
  try {
    final boundary =
        _cardKey.currentContext?.findRenderObject() as RenderRepaintBoundary?;
    if (boundary == null) return null;

    final ui.Image image =
        await boundary.toImage(pixelRatio: ui.window.devicePixelRatio);
    final byteData = await image.toByteData(format: ui.ImageByteFormat.png);
    if (byteData == null) return null;

    final pngBytes = byteData.buffer.asUint8List();

    if (kIsWeb) {
      // For Web: download the file directly in the browser
      final blob = html.Blob([pngBytes], 'image/png');
      final url = html.Url.createObjectUrlFromBlob(blob);
      final anchor = html.AnchorElement(href: url)
        ..download = 'id_card_${DateTime.now().millisecondsSinceEpoch}.png'
        ..click();
      html.Url.revokeObjectUrl(url);
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('‚úÖ Download started in browser')),
      );
      return null;
    } else {
      // For Mobile/Desktop
      final dir = await getApplicationDocumentsDirectory();
      final file = File(
          '${dir.path}/id_card_${DateTime.now().millisecondsSinceEpoch}.png');
      await file.writeAsBytes(pngBytes);
      return file;
    }
  } catch (e) {
    debugPrint('Export error: $e');
    return null;
  }
}

class _CounterWidgetState extends State<CounterWidget> {

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      floatingActionButton: Column(
        mainAxisAlignment: MainAxisAlignment.end,

              setState(() {
                counter--;

      ),

        child: Text(
          '$counter',
          style: const TextStyle(
            fontSize: 30,

    );
